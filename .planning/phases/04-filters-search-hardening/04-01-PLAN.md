    Then create `app/lib/filters.ts` exporting:

    ```typescript
    export interface FilterState {
      statuses: Set<TaskStatus>       // empty = show all
      requestTypes: Set<RequestType>  // empty = show all
      dateFrom: string | null         // ISO date string YYYY-MM-DD
      dateTo: string | null           // ISO date string YYYY-MM-DD
      search: string                  // substring match
      sortDirection: 'asc' | 'desc'   // asc = oldest first (default)
    }

    export const DEFAULT_FILTERS: FilterState = {
      statuses: new Set(),
      requestTypes: new Set(),
      dateFrom: null,
      dateTo: null,
      search: '',
      sortDirection: 'asc',
    }

    export function applyFilters(tasks: TaskWithStaff[], filters: FilterState): TaskWithStaff[]
    ```

    Implementation rules per D-02: filters combine with AND logic. Per research pitfall 2: empty Set means "show all" not "show none" -- check `size > 0` before filtering. Per pitfall 3: dateTo should add one day for exclusive upper bound to include the full end date. Search is case-insensitive substring on client_name or phone per D-07.
